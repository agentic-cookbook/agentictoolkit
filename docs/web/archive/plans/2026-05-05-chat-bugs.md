# Chat Bug Sweep Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Resolve a batch of correctness, layout, and theming bugs across the chat package and the chat demo, plus a small architectural change to the connector system and the mobile-overlay component.

**Architecture:** Most fixes are localized CSS / small component edits. Two structural changes:
1. `MobileChat` becomes `ContentOverlay` — a generic overlay shell that wraps any chat (or any content). The chat demo treats it as an additive boolean toggle, not a mode swap.
2. Connector lines stop measuring arbitrary chat DOM. Instead, components render explicit `<ConnectorAnchor>` markers (themable via CSS); a `ConnectorRegistry` tracks anchor positions via `ResizeObserver`/`MutationObserver` and the SVG draws between registered anchors.

In addition the three-pane layout becomes group-centered (no fixed bottom-right), the topic list moves to the left, and panels become a stack that reflows when new ones open.

**Tech Stack:** React 19, TypeScript, Tailwind 4, Vite, Vitest + @testing-library/react + jsdom.

Goal: when a message has unloaded images, hide those images (don't reserve a broken/empty box) and keep the typing indicator visible for the message until all its images are loaded. The simplest mechanism: each `<img>` is rendered with `style={{ display: loaded ? '' : 'none' }}` and the message-level "is-loading" boolean is computed from a count of `loaded === total`. Decision (state explicitly here so the implementer doesn't re-decide): **A message that has any image content does not render until *all* of its images have loaded.** Until then the global typing indicator at the transcript bottom keeps showing (already its purpose). import { useEffect, useState } from 'react'
```tsx
import type { ContentItem } from '../types'

export function useAllImagesLoaded(content?: ContentItem[]): boolean {
  const imageSrcs = (content ?? [])
    .filter((c): c is { type: 'image'; src: string } => c.type === 'image')
    .map((c) => c.src)
  const [loaded, setLoaded] = useState<Record<string, boolean>>({})
  useEffect(() => {
    setLoaded({})
    imageSrcs.forEach((src) => {
      const img = new Image()
      img.onload = () => setLoaded((p) => ({ ...p, [src]: true }))
      img.onerror = () => setLoaded((p) => ({ ...p, [src]: true })) // fail-open
      img.src = src
    })
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [imageSrcs.join('|')])
  return imageSrcs.every((s) => loaded[s])
}
```

- [ ] **Step 3: Implement `ConnectorRegistry.tsx`**

```tsx
import {
  createContext,
  useCallback,
  useContext,
  useMemo,
  useRef,
  useSyncExternalStore,
  type ReactNode,
} from 'react'

type AnchorMap = Map<string, HTMLElement>

interface RegistryHandle {
  register(id: string, el: HTMLElement): () => void
  snapshot(): { has(id: string): boolean; get(id: string): HTMLElement | undefined; ids(): string[] }
  subscribe(listener: () => void): () => void
}

const Ctx = createContext<RegistryHandle | null>(null)

export function ConnectorRegistryProvider({ children }: { children: ReactNode }) {
  const map = useRef<AnchorMap>(new Map())
  const listeners = useRef<Set<() => void>>(new Set())

  const handle = useMemo<RegistryHandle>(() => {
    const notify = () => listeners.current.forEach((fn) => fn())
    return {
      register(id, el) {
        map.current.set(id, el)
        notify()
        return () => {
          if (map.current.get(id) === el) {
            map.current.delete(id)
            notify()
          }
        }
      },
      snapshot() {
        const m = map.current
        return {
          has: (id) => m.has(id),
          get: (id) => m.get(id),
          ids: () => Array.from(m.keys()),
        }
      },
      subscribe(listener) {
        listeners.current.add(listener)
        return () => listeners.current.delete(listener)
      },
    }
  }, [])

  return <Ctx.Provider value={handle}>{children}</Ctx.Provider>
}

export function useConnectorRegistry(): RegistryHandle {
  const v = useContext(Ctx)
  if (!v) throw new Error('useConnectorRegistry must be used inside <ConnectorRegistryProvider>')
  return v
}

// Subscribe-aware snapshot (for the SVG)
export function useRegistrySnapshot() {
  const reg = useConnectorRegistry()
  const subscribe = useCallback((l: () => void) => reg.subscribe(l), [reg])
  const get = useCallback(() => reg.snapshot(), [reg])
  return useSyncExternalStore(subscribe, get, get)
}
```

### Task 10: New `ConnectorSVG` driven by registry + ResizeObserver

**Files:**
- Modify: `packages/chat/modes/three-pane/ConnectorSVG.tsx`
- Modify: `packages/chat/modes/ThreePaneChat.tsx`
- Modify: `packages/chat/css/modes/three-pane.css` (default anchor styles)

The new `ConnectorSVG` accepts a list of `{ from: string; to: string }` pairs (anchor ids) and a `frameRef` for coordinate-space conversion. It uses `useRegistrySnapshot` plus a `ResizeObserver` watching every registered anchor and the frame to redraw whenever anything moves.

- [ ] **Step 1: Replace `ConnectorSVG.tsx`**

```tsx
import { useEffect, useRef, useState, type RefObject } from 'react'
import { useConnectorRegistry, useRegistrySnapshot } from './connectors/ConnectorRegistry'

export interface ConnectorPair {
  from: string
  to: string
}

export interface ConnectorSVGProps {
  frameRef: RefObject<HTMLDivElement | null>
  pairs: ConnectorPair[]
}

interface Geom { x1: number; y1: number; x2: number; y2: number; color: string }

export function ConnectorSVG({ frameRef, pairs }: ConnectorSVGProps) {
  const reg = useConnectorRegistry()
  const snap = useRegistrySnapshot()
  const [geoms, setGeoms] = useState<Geom[]>([])
  const rafRef = useRef<number | null>(null)

  useEffect(() => {
    const frame = frameRef.current
    if (!frame) return

    const compute = () => {
      const fb = frame.getBoundingClientRect()
      const out: Geom[] = []
      for (const p of pairs) {
        const a = snap.get(p.from)
        const b = snap.get(p.to)
        if (!a || !b) continue
        const ar = a.getBoundingClientRect()
        const br = b.getBoundingClientRect()
        const color = getComputedStyle(a).color
        out.push({
          x1: ar.left + ar.width / 2 - fb.left,
          y1: ar.top + ar.height / 2 - fb.top,
          x2: br.left + br.width / 2 - fb.left,
          y2: br.top + br.height / 2 - fb.top,
          color,
        })
      }
      setGeoms(out)
    }

    const schedule = () => {
      if (rafRef.current != null) return
      rafRef.current = requestAnimationFrame(() => {
        rafRef.current = null
        compute()
      })
    }

    const ro = new ResizeObserver(schedule)
    ro.observe(frame)
    const watched: Element[] = []
    for (const id of snap.ids()) {
      const el = snap.get(id)
      if (el) {
        ro.observe(el)
        watched.push(el)
      }
    }
    window.addEventListener('resize', schedule)
    window.addEventListener('scroll', schedule, true)
    schedule()

    return () => {
      ro.disconnect()
      window.removeEventListener('resize', schedule)
      window.removeEventListener('scroll', schedule, true)
      if (rafRef.current != null) cancelAnimationFrame(rafRef.current)
    }
    // re-run when pairs or registry membership changes
  }, [pairs, snap, frameRef, reg])

  return (
    <svg className="pc-connector-svg" aria-hidden="true">
      {geoms.map((g, i) => (
        <g key={i}>
          <line
            x1={g.x1}
            y1={g.y1}
            x2={g.x2}
            y2={g.y2}
            stroke={g.color}
            strokeOpacity={0.4}
            strokeWidth={1}
          />
          <circle cx={g.x1} cy={g.y1} r={3} fill={g.color} fillOpacity={0.4} />
          <circle cx={g.x2} cy={g.y2} r={3} fill={g.color} fillOpacity={0.4} />
        </g>
      ))}
    </svg>
  )
}
```
